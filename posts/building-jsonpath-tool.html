从零实现 JSONPath 查询工具

编译器前端实战：词法分析、语法解析、AST 遍历

📅 2026-03-18 ⏱️ 阅读时间: 15 分钟 🎯 难度: 中级

📖 前言

你有没有想过，像 jq 或 jsonpath 这样的工具是如何工作的？ 它们是如何将一个简单的查询字符串（如 .users[0].name）转换成对 JSON 数据的精确访问？

在这篇教程中，我们将从零开始实现一个简单的 JSONPath 查询工具。 通过这个项目，你将学到编译器前端的核心概念：

  • 词法分析（Lexing）：将字符串转换为 Token 流
  • 语法分析（Parsing）：将 Token 流转换为抽象语法树（AST）
  • 解释执行（Interpretation）：在 JSON 数据上执行 AST
💡 项目源码： https://github.com/robertsong2019/jp
核心代码 < 500 行，适合学习和理解编译器基础

🏗️ 整体架构

我们的 JSONPath 工具由三个核心模块组成：

jp/
├── lexer.py      # 词法分析器 - Token 化
├── parser.py     # 解析器 - 构建 AST
├── query.py      # 查询引擎 - 执行查询
└── cli.py        # 命令行接口

数据流向如下：

路径字符串 → Lexer → Token 流 → Parser → AST → Query Engine → 结果

第一步：词法分析（Lexer）

词法分析器的作用是将输入字符串转换为一系列有意义的 Token。 每个Token代表一个最小的语法单元。

定义 Token 类型

from dataclasses import dataclass
from enum import Enum, auto

class TokenType(Enum):
    DOT = auto()          # .
    LBRACKET = auto()     # [
    RBRACKET = auto()     # ]
    NUMBER = auto()       # 123
    IDENTIFIER = auto()   # name, users, etc.
    WILDCARD = auto()     # * or [*]
    EOF = auto()          # End of input

@dataclass
class Token:
    type: TokenType
    value: any
    position: int

实现 Lexer

class Lexer:
    def __init__(self, text: str):
        self.text = text
        self.pos = 0
        self.current_char = text[0] if text else None

    def advance(self):
        """移动到下一个字符"""
        self.pos += 1
        self.current_char = self.text[self.pos] if self.pos < len(self.text) else None

    def skip_whitespace(self):
        """跳过空白字符"""
        while self.current_char and self.current_char.isspace():
            self.advance()

    def number(self) -> Token:
        """解析数字"""
        start_pos = self.pos
        result = ''
        while self.current_char and self.current_char.isdigit():
            result += self.current_char
            self.advance()
        return Token(TokenType.NUMBER, int(result), start_pos)

    def identifier(self) -> Token:
        """解析标识符"""
        start_pos = self.pos
        result = ''
        while self.current_char and (self.current_char.isalnum() or self.current_char == '_'):
            result += self.current_char
            self.advance()
        return Token(TokenType.IDENTIFIER, result, start_pos)

    def get_next_token(self) -> Token:
        """获取下一个 Token"""
        while self.current_char:
            if self.current_char.isspace():
                self.skip_whitespace()
                continue

            if self.current_char == '.':
                self.advance()
                return Token(TokenType.DOT, '.', self.pos - 1)

            if self.current_char == '[':
                self.advance()
                return Token(TokenType.LBRACKET, '[', self.pos - 1)

            if self.current_char == ']':
                self.advance()
                return Token(TokenType.RBRACKET, ']', self.pos - 1)

            if self.current_char == '*':
                self.advance()
                return Token(TokenType.WILDCARD, '*', self.pos - 1)

            if self.current_char.isdigit():
                return self.number()

            if self.current_char.isalpha() or self.current_char == '_':
                return self.identifier()

            raise Exception(f'Invalid character: {self.current_char}')

        return Token(TokenType.EOF, None, self.pos)

测试 Lexer

# 输入: .users[0].name
lexer = Lexer('.users[0].name')
tokens = []
while True:
    token = lexer.get_next_token()
    tokens.append(token)
    if token.type == TokenType.EOF:
        break

# 输出:
# [DOT, IDENTIFIER("users"), LBRACKET, NUMBER(0), RBRACKET, DOT, IDENTIFIER("name"), EOF]

第二步：语法分析（Parser）

语法分析器将 Token 流转换为 抽象语法树（AST）。 AST 是查询语句的结构化表示。

定义 AST 节点

@dataclass
class FieldNode:
    """字段访问节点，如 .name"""
    name: str

@dataclass
class IndexNode:
    """数组索引节点，如 [0]"""
    index: int

@dataclass
class WildcardNode:
    """通配符节点，如 .* 或 [*]"""
    pass

实现 Parser

class Parser:
    def __init__(self, lexer: Lexer):
        self.lexer = lexer
        self.current_token = self.lexer.get_next_token()

    def eat(self, token_type: TokenType):
        """消耗当前 Token，并获取下一个"""
        if self.current_token.type == token_type:
            self.current_token = self.lexer.get_next_token()
        else:
            raise Exception(f'Expected {token_type}, got {self.current_token.type}')

    def parse(self) -> list:
        """解析 JSONPath 表达式，返回 AST 节点列表"""
        ast = []

        # 跳过可选的根节点标识符 $
        if self.current_token.type == TokenType.IDENTIFIER and self.current_token.value == '$':
            self.eat(TokenType.IDENTIFIER)

        while self.current_token.type != TokenType.EOF:
            if self.current_token.type == TokenType.DOT:
                self.eat(TokenType.DOT)
                if self.current_token.type == TokenType.IDENTIFIER:
                    ast.append(FieldNode(self.current_token.value))
                    self.eat(TokenType.IDENTIFIER)
                elif self.current_token.type == TokenType.WILDCARD:
                    ast.append(WildcardNode())
                    self.eat(TokenType.WILDCARD)

            elif self.current_token.type == TokenType.LBRACKET:
                self.eat(TokenType.LBRACKET)
                if self.current_token.type == TokenType.NUMBER:
                    ast.append(IndexNode(self.current_token.value))
                    self.eat(TokenType.NUMBER)
                elif self.current_token.type == TokenType.WILDCARD:
                    ast.append(WildcardNode())
                    self.eat(TokenType.WILDCARD)
                self.eat(TokenType.RBRACKET)

        return ast

测试 Parser

# 输入: .users[0].name
lexer = Lexer('.users[0].name')
parser = Parser(lexer)
ast = parser.parse()

# 输出 AST:
# [
#   FieldNode("users"),
#   IndexNode(0),
#   FieldNode("name")
# ]

第三步：查询执行（Query Engine）

查询引擎在 JSON 数据上遍历 AST，返回匹配的值。

实现 Query Engine

class QueryEngine:
    def __init__(self, data: any):
        self.data = data

    def execute(self, ast: list) -> list:
        """执行 AST，返回所有匹配的值"""
        results = [self.data]

        for node in ast:
            new_results = []
            for result in results:
                if isinstance(node, FieldNode):
                    if isinstance(result, dict) and node.name in result:
                        new_results.append(result[node.name])
                elif isinstance(node, IndexNode):
                    if isinstance(result, list) and 0 <= node.index < len(result):
                        new_results.append(result[node.index])
                elif isinstance(node, WildcardNode):
                    if isinstance(result, dict):
                        new_results.extend(result.values())
                    elif isinstance(result, list):
                        new_results.extend(result)
            results = new_results

        return results

完整示例

import json

# JSON 数据
data = {
    "users": [
        {"name": "Alice", "age": 30},
        {"name": "Bob", "age": 25}
    ]
}

# 查询: .users[0].name
path = '.users[0].name'

# 词法分析
lexer = Lexer(path)
tokens = []
while True:
    token = lexer.get_next_token()
    tokens.append(token)
    if token.type == TokenType.EOF:
        break

# 语法分析
lexer = Lexer(path)  # 重新创建 lexer
parser = Parser(lexer)
ast = parser.parse()

# 执行查询
engine = QueryEngine(data)
results = engine.execute(ast)

# 输出: ["Alice"]
print(results)

第四步：命令行接口（CLI）

最后，我们添加一个命令行接口，让用户可以方便地使用我们的工具。 

import argparse
import sys
import json

def main():
    parser = argparse.ArgumentParser(description='JSONPath query tool')
    parser.add_argument('path', help='JSONPath expression')
    parser.add_argument('-j', '--json', action='store_true',
                       help='Output as JSON')
    parser.add_argument('-r', '--raw', action='store_true',
                       help='Raw output (no quotes for strings)')
    parser.add_argument('--version', action='version', version='jp 1.0.0')

    args = parser.parse_args()

    # 从 stdin 读取 JSON
    try:
        data = json.load(sys.stdin)
    except json.JSONDecodeError as e:
        print(f'Error: Invalid JSON input: {e}', file=sys.stderr)
        sys.exit(1)

    # 解析和执行查询
    try:
        lexer = Lexer(args.path)
        parser_obj = Parser(lexer)
        ast = parser_obj.parse()
        engine = QueryEngine(data)
        results = engine.execute(ast)

        # 输出结果
        for result in results:
            if args.json:
                print(json.dumps(result))
            elif args.raw and isinstance(result, str):
                print(result)
            else:
                print(json.dumps(result) if not isinstance(result, str) else f'"{result}"')

    except Exception as e:
        print(f'Error: {e}', file=sys.stderr)
        sys.exit(1)

if __name__ == '__main__':
    main()

使用示例

# 简单字段访问
echo '{"name": "Alice", "age": 30}' | jp .name
# 输出: "Alice"

# 数组索引
echo '{"items": ["apple", "banana"]}' | jp '.items[1]'
# 输出: "banana"

# 嵌套对象
echo '{"user": {"name": "Bob"}}' | jp .user.name
# 输出: "Bob"

# 通配符
echo '{"users": [{"name": "Alice"}, {"name": "Bob"}]}' | jp '.users[*].name'
# 输出:
# "Alice"
# "Bob"
